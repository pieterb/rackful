def header
  retval = super
  retval += "<h1>HTTP/1.1 #{Rack::Utils.escape_html(resource.title)}</h1>\n"
  unless resource.message.empty?
    retval += "<div id=\"rackful-description\">#{resource.message}</div>\n"
  end
  retval
end
