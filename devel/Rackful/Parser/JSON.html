def recursive_datetime_parser p
  if p.kind_of?(String)
    begin
      return Time.xmlschema(p)
    rescue
    end
  elsif p.kind_of?(Hash)
    p.keys.each do
      |key|
      p[key] = self.recursive_datetime_parser( p[key] )
    end
  elsif p.kind_of?(Array)
    (0 ... p.size).each do
      |i|
      p[i] = self.recursive_datetime_parser( p[i] )
    end
  end
  p
end
