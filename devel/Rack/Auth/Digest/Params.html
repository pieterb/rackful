def self.dequote(str) # From WEBrick::HTTPUtils
  ret = (/\A"(.*)"\Z/ =~ str) ? $1 : str.dup
  ret.gsub!(/\\(.)/, "\\1")
  ret
end

def self.parse(str)
  Params[*split_header_value(str).map do |param|
    k, v = param.split('=', 2)
    [k, dequote(v)]
  end.flatten]
end

def self.split_header_value(str)
  str.scan( /(\w+\=(?:"[^\"]+"|[^,]+))/n ).collect{ |v| v[0] }
end

def to_s
  map do |k, v|
    "#{k}=" + (UNQUOTED.include?(k) ? v.to_s : quote(v))
  end.join(', ')
end
