def each
  term = TERM
  @body.each do |chunk|
    size = bytesize(chunk)
    next if size == 0

    chunk = chunk.dup.force_encoding(Encoding::BINARY) if chunk.respond_to?(:force_encoding)
    yield [size.to_s(16), term, chunk, term].join
  end
  yield TAIL
end
