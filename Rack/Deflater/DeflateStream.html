def each
  deflater = ::Zlib::Deflate.new(*DEFLATE_ARGS)
  @body.each { |part| yield deflater.deflate(part, Zlib::SYNC_FLUSH) }
  yield deflater.finish
  nil
ensure
  @body.close if @body.respond_to?(:close)
  deflater.close
end
